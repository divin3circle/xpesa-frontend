import { creatorStartUrl } from '/snippets/config.mdx';

<Steps>
  <Step title="Open the creator signup page">
    Go to <a href={creatorStartUrl}>the creator waitlist</a> and enter your name, email, creator focus, and what you want to sell first.
  </Step>
  <Step title="Verify your email">
    If email verification is required, open the verification message and confirm your address.
  </Step>
  <Step title="Continue onboarding">
    After verification, xpesa guides you through wallet setup, handle selection, and profile setup.
  </Step>
  <Step title="Create your first paid experience">
    Once setup is complete, create a gate link, document, file pack, or tip link.
  </Step>
</Steps>

<Tip>
  Use the same email you want to receive creator updates and account notifications on.
</Tip>
